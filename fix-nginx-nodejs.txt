# Fix Nginx to properly serve Node.js application
# Run these commands in your SSH terminal:

# 1. Create proper Nginx site configuration
cat > /etc/nginx/sites-available/ienet.online << 'EOF'
server {
    listen 80;
    server_name ienet.online www.ienet.online;
    
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
    
    # Static files fallback
    location /static/ {
        root /var/www/vhosts/vivaindia.com/ienet.online/dist/public;
        try_files $uri $uri/ =404;
    }
}
EOF

# 2. Enable the site
ln -sf /etc/nginx/sites-available/ienet.online /etc/nginx/sites-enabled/

# 3. Remove default site if it exists
rm -f /etc/nginx/sites-enabled/default

# 4. Test Nginx configuration
nginx -t

# 5. Restart Nginx
systemctl restart nginx

# 6. Navigate to application directory
cd /var/www/vhosts/vivaindia.com/ienet.online

# 7. Create production environment file
cat > .env << 'EOF'
NODE_ENV=production
PORT=3000
DATABASE_URL=mysql://netiedb:h5pLF9833@localhost:3306/ienetdb
DB_HOST=localhost
DB_PORT=3306
DB_NAME=ienetdb
DB_USER=netiedb
DB_PASSWORD=h5pLF9833
DOMAIN=ienet.online
EOF

# 8. Install PM2 for process management
npm install -g pm2

# 9. Start the application with PM2
pm2 start npm --name "ienet" -- start

# 10. Save PM2 configuration
pm2 save
pm2 startup

# 11. Check if application is running
pm2 status
curl -I http://localhost:3000

# 12. Test the website
curl -I http://ienet.online
