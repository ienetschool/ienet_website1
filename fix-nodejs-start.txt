NODEJS STARTUP COMMANDS FOR YOUR SERVER:

# Navigate to application directory
cd /ienet.online

# Install dependencies
npm install

# Start application (choose one method):

# Method 1: Direct start
node index.js

# Method 2: Using PM2 (recommended for production)
npm install -g pm2
pm2 start index.js --name ienet-app
pm2 save
pm2 startup

# Method 3: Using systemd service
sudo tee /etc/systemd/system/ienet.service > /dev/null <<EOF
[Unit]
Description=IeNet React Application
After=network.target

[Service]
Type=simple
User=www-data
WorkingDirectory=/ienet.online
ExecStart=/usr/bin/node index.js
Restart=always
RestartSec=10
Environment=NODE_ENV=production
Environment=PORT=3000

[Install]
WantedBy=multi-user.target
EOF

sudo systemctl daemon-reload
sudo systemctl enable ienet
sudo systemctl start ienet

# Check status
sudo systemctl status ienet

# View logs
sudo journalctl -u ienet -f
